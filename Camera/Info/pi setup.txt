---------------Use 32 bit buster (debian 10) image---------------------

---------------install ros noetic----------------------
sudo sh -c 'echo "deb http://packages.ros.org/ros/ubuntu $(lsb_release -sc) main" > /etc/apt/sources.list.d/ros-latest.list'
sudo apt-key adv --keyserver hkp://keyserver.ubuntu.com:80 --recv-key C1CF6E31E6BADE8868B172B4F42ED6FBAB17C654
sudo apt update
sudo apt upgrade

sudo apt-get install python3-rosdep python3-rosinstall-generator python3-vcstools python3-vcstool build-essential
sudo rosdep init
rosdep update
mkdir ~/ros_catkin_ws
cd ~/ros_catkin_ws
sudo pip3 install -U rospkg
sudo pip3 install -U rosdistro
rosinstall_generator desktop --rosdistro noetic --deps --tar > noetic-desktop.rosinstall
mkdir ./src
vcs import --input noetic-desktop.rosinstall ./src
rosdep install --from-paths ./src --ignore-packages-from-source --rosdistro noetic -y
./src/catkin/bin/catkin_make_isolated --install -DCMAKE_BUILD_TYPE=Release -DPYTHON_EXECUTABLE=/usr/bin/python3

echo "source ~/ros_catkin_ws/install_isolated/setup.bash" >> ~/.bashrc
echo "alias eb='nano ~/.bashrc'" >> ~/.bashrc
echo "alias sb='source ~/.bashrc'" >> ~/.bashrc

source ~/.bashrc


---------------install catkin/make workspace----------------------
sudo apt-get install cmake python-catkin-pkg python-empy python-nose python-setuptools libgtest-dev build-essential python-catkin-sphinx
cd ~
git clone https://github.com/ros/catkin.git
cd catkin
mkdir build && cd build && cmake ../ && make && sudo make install -DCMAKE_BUILD_TYPE=Release

sudo apt-get install python-catkin-tools
mkdir -p ~/catkin_ws/src
cd ~/catkin_ws/
catkin build

echo "alias cw='cd ~/catkin_ws'" >> ~/.bashrc
echo "alias cb='cw && catkin_build'" >> ~/.bashrc

echo "source ~/catkin_ws/devel/setup.bash" >> ~/.bashrc

source ~/.bashrc

cd ~/catkin_ws/src
catkin_create_pkg needle_insertion_robot std_msgs rospy roscpp cv_bridge geometry_msgs sensor_msgs
cb
sb
cd ~/catkin_ws/src/needle_insertion_robot/
mkdir scripts launch


---------------program dependencies--------------------
enable camera in "sudo raspi-config" under interface options
pip3 install pigpio
pip3 install picamera
